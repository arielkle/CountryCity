        function playTick(freq = 250, duration = 0.05) {

            if (audioCtx.state === 'suspended') return;

            const osc = audioCtx.createOscillator();

            const gain = audioCtx.createGain();

            osc.type = 'sine';

            osc.frequency.setValueAtTime(freq, audioCtx.currentTime);

            gain.gain.setValueAtTime(0.05, audioCtx.currentTime);

            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);

            osc.connect(gain);

            gain.connect(audioCtx.destination);

            osc.start();

            osc.stop(audioCtx.currentTime + duration);

        }

        function handleStateUpdate(data) {

                case 'waiting':

                    resetModals();

                    document.getElementById('result').innerHTML = "";

                    hasAutoGradedThisRound = false;

                    updateTeacherSpinButton();

                    break;

                

                case 'spinning':

                    resetModals();

                    if (currentUser && currentUser.role === 'teacher') {

                        document.getElementById('spinBtn').disabled = true;

                        document.getElementById('spinBtn').innerText = "מגריל...";

                        document.getElementById('forceSpinBtn').classList.add('hidden');

                    }

                    document.getElementById('result').innerHTML = "";

                    

                    if (audioCtx.state === 'suspended') audioCtx.resume();

                    document.getElementById('pointer').classList.add("pointer-active");

                    

                    // סיבוב ויזואלי לכולם

                    canvas.style.transform = `rotate(${data.spinTarget}deg)`;

                    

                    // צליל סיבוב מקומי

                    if (tickInterval) clearInterval(tickInterval);

                    tickInterval = setInterval(() => playTick(280), 160);

                    setTimeout(() => {

                        clearInterval(tickInterval);

                        document.getElementById('pointer').classList.remove("pointer-active");

                    document.getElementById('result').innerHTML = `<span class="letter-winner text-indigo-600">${data.currentLetter}</span>`;

                        openAnswerModal(data.currentLetter, data.categories);

                    }

                    break;


                case 'evaluating':

                    if (currentUser.role === 'teacher') {

                        // המורה בודק את המנצח

                        openEvaluationModal(data.winnerData);

        function drawWheel(alphabet) {

            const centerX = canvas.width / 2;

            const centerY = canvas.height / 2;

            const radius = centerX - 15;

            ctx.clearRect(0, 0, canvas.width, canvas.height);


            if (alphabet.length === 0) {

                ctx.fillStyle = "#cbd5e1";

                ctx.beginPath(); ctx.arc(centerX, centerY, radius, 0, Math.PI * 2); ctx.fill();

                ctx.fillStyle = "white"; ctx.textAlign = "center"; ctx.font = "bold 24px Assistant";

                ctx.fillText("נגמרו האותיות", centerX, centerY);

                return;

            }


            const step = (Math.PI * 2) / alphabet.length;

            alphabet.forEach((letter, i) => {

                const angle = i * step;

                ctx.beginPath();

                ctx.fillStyle = wheelColors[i % wheelColors.length];

                ctx.moveTo(centerX, centerY);

                ctx.arc(centerX, centerY, radius, angle, angle + step);

                ctx.fill();

                

                ctx.save();

                ctx.translate(centerX, centerY);

                ctx.rotate(angle + step / 2);

                ctx.textAlign = "right";

                ctx.fillStyle = "white";

                const fontSize = alphabet.length > 15 ? 32 : (alphabet.length > 5 ? 40 : 55);

                ctx.font = `bold ${fontSize}px Assistant`;

                ctx.fillText(letter, radius - 35, 12);

                ctx.restore();

            });


            // עיגול פנימי

            const grad = ctx.createRadialGradient(centerX, centerY, 0, centerX, centerY, 45);

            grad.addColorStop(0, '#ffffff'); grad.addColorStop(1, '#f1f5f9');

            ctx.beginPath(); ctx.arc(centerX, centerY, 45, 0, Math.PI * 2);

            ctx.fillStyle = grad; ctx.fill();

        }

                answers.push({ category: catName, value: val || '-' });

            });

                await updateDoc(stateRef, {

                    status: 'evaluating',

                        name: currentUser.name,

                        answers: answers

                    }

                });

            }

        };

        async function autoGradeMyIncompleteForm() {

            if (hasAutoGradedThisRound) return;

            hasAutoGradedThisRound = true;


            // ספירת השדות הלא ריקים

            const inputs = document.querySelectorAll('.answer-input');

            let pointsEarned = 0;

            inputs.forEach(input => {

                if (input.value.trim() !== '') pointsEarned++;

            });


            if (pointsEarned > 0) {

                await updateDoc(getMyPlayerRef(), { score: increment(pointsEarned) });

            }

        }
